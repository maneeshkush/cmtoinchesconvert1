2 ایم پی ایس کو واٹ میں کیسے تبدیل کریں۔

2 amps (A) کے برقی رو کو واٹ (W) میں برقی طاقت میں تبدیل کرنے کا طریقہ۔

آپ ایم پی ایس اور وولٹ سے واٹ کا حساب لگا سکتے ہیں (لیکن تبدیل نہیں)

12V DC کے وولٹیج کے ساتھ 2A سے واٹ کا حساب

ڈی سی سرکٹ میں، پاور (واٹ میں) کرنٹ کے برابر ہوتی ہے (ایمپیئر میں) وولٹیج (وولٹ میں) سے ضرب۔لہذا اگر آپ ڈی سی سرکٹ میں کرنٹ اور وولٹیج کو جانتے ہیں، تو آپ واٹس میں پاور کا حساب لگانے کے لیے یہ فارمولہ استعمال کر سکتے ہیں:

watts = amps × volts

مثال کے طور پر، اگر آپ کے پاس DC پاور سپلائی ہے جس کا وولٹیج 12V ہے اور کرنٹ 2A ہے، تو پاور یہ ہوگی:

watts = 2A × 12V = 24W

یہ جاننے کے لیے ایک بہت مفید فارمولا ہے، کیونکہ یہ آپ کو کسی ڈیوائس کی بجلی کی کھپت یا پاور سپلائی کے پاور آؤٹ پٹ کا حساب لگانے کی اجازت دیتا ہے۔یہ تار اور دیگر اجزاء کے سائز کا تعین کرنے کے لیے بھی مفید ہے جو آپ کو ڈی سی سرکٹ میں استعمال کرنے کی ضرورت ہے۔

120V AC کے وولٹیج کے ساتھ 2A سے واٹ کا حساب

ایک AC سرکٹ میں، پاور (واٹ میں) پاور فیکٹر (PF) کے برابر ہوتی ہے جو کرنٹ (ایمپیئرز میں) کو وولٹیج (وولٹ میں) سے ضرب دی جاتی ہے۔پاور فیکٹر اس بات کا ایک پیمانہ ہے کہ سرکٹ میں برقی طاقت کا استعمال کتنے مؤثر طریقے سے ہو رہا ہے۔یہ ایک یونٹ لیس قدر ہے جو 0 سے 1 تک ہو سکتی ہے، 1 ایک کامل پاور فیکٹر ہونے کے ساتھ۔

اگر آپ کے پاس AC پاور سپلائی ہے جس کا وولٹیج 120V ہے اور کرنٹ 2A ہے، اور لوڈ ایک مزاحمتی بوجھ ہے (جیسے ہیٹنگ عنصر)، پاور فیکٹر 1 ہوگا، اور پاور یہ ہوگی:

watts = 1 × 2A × 120V = 240W

اگر بوجھ ایک آمادہ بوجھ ہے (جیسے انڈکشن موٹر)، تو پاور فیکٹر 1 سے کم ہوگا، عام طور پر 0.8 کے آس پاس۔اس صورت میں، طاقت ہوگی:

watts = 0.8 × 2A × 120V = 192W

AC سرکٹ میں پاور کا حساب لگاتے وقت پاور فیکٹر پر غور کرنا ضروری ہے، کیونکہ یہ تار اور دیگر اجزاء کے سائز کو متاثر کر سکتا ہے جو آپ کو سرکٹ میں استعمال کرنے کی ضرورت ہے۔مثال کے طور پر، اگر پاور فیکٹر کم ہے، تو آپ کو زیادہ کرنٹ کو سنبھالنے کے لیے بڑی تار یا دیگر اجزاء استعمال کرنے کی ضرورت پڑ سکتی ہے جس کی اتنی ہی مقدار میں بجلی فراہم کرنے کی ضرورت ہے۔

230V AC کے وولٹیج کے ساتھ 2A سے واٹ کا حساب

اگر آپ کے پاس AC پاور سپلائی ہے جس کا وولٹیج 230V اور کرنٹ 2A ہے، اور لوڈ ایک مزاحمتی بوجھ ہے (جیسے حرارتی عنصر)، پاور فیکٹر 1 ہوگا، اور پاور یہ ہوگی:

watts = 1 × 2A × 230V = 460W

اگر بوجھ ایک آمادہ بوجھ ہے (جیسے انڈکشن موٹر)، تو پاور فیکٹر 1 سے کم ہوگا، عام طور پر 0.8 کے آس پاس۔اس صورت میں، طاقت ہوگی:

watts = 0.8 × 2A × 230V = 368W

یہ نوٹ کرنا ضروری ہے کہ AC سرکٹ میں وولٹیج اور موجودہ قدریں مستقل نہیں ہیں، بلکہ وہ وقت کے ساتھ ساتھ سائنوسائڈ طور پر مختلف ہوتی ہیں۔پاور فیکٹر اس بات کا ایک پیمانہ ہے کہ سرکٹ میں برقی طاقت کا استعمال کتنے مؤثر طریقے سے ہو رہا ہے، اور یہ وولٹیج اور کرنٹ ویوفارمز کے درمیان فیز اینگل سے متاثر ہو سکتا ہے۔عام طور پر، زیادہ پاور فیکٹر اشارہ کرتا ہے کہ برقی طاقت زیادہ مؤثر طریقے سے استعمال ہو رہی ہے، جب کہ کم پاور فیکٹر اشارہ کرتا ہے کہ کچھ پاور ضائع ہو رہی ہے۔

ایمپیئر کی پیمائش کیا ہے اور 1 ایمپیئر سے واٹ کی پیمائش کیسے کی جائے؟

ایمپیئر کو ایممیٹر سے ماپا جاتا ہے۔کرنٹ کی اکائی ایمپیئر کہلاتی ہے۔جس طرح ہم وزن کو کلوگرام میں ماپتے ہیں، لمبائی فٹ یا میٹر میں ناپی جاتی ہے، اسی طرح کرنٹ کو ایمپیئر میں ناپا جاتا ہے۔ 

Ammeter سرکٹس کی سیریز میں رکھا جاتا ہے۔اور ہم اسے آن کر کے کرنٹ کی قدر معلوم کر سکتے ہیں۔ 

اگر ہم سرکٹ میں مزاحمت اور وولٹیج کا دماغ جانتے ہیں، تو ہم اوہم کے قانون (V = IR) سے کرنٹ یا 1 ایمپیئر تا واٹ کے دماغ کی قدر تلاش کر سکتے ہیں۔ 

1 ایمپیئر کی تعریف 1 ایمپیئر کی تعریف 

1 ایمپیئر سے واٹ کو سمجھنے سے پہلے  ، ہم ایمپیئر کی تعریف کو سمجھتے ہیں کیونکہ ہم جانتے ہیں کہ 1 کولمب چارج میں الیکٹران کی تعداد 6.25 × 10¹⁸ ہے۔ 

1 A کی تعریف: جب  1 کلوم چارج برقی سرکٹ میں 1 سیکنڈ میں بہتا ہے، تو سرکٹ میں بہنے والے کرنٹ کی قدر 1 ایمپیئر ہوگی۔ 

یعنی، جب ایک سرکٹ میں 1 سیکنڈ میں الیکٹران کا بہاؤ 6.25×10¹¹⁸ ہوتا ہے، تو کرنٹ کے بہاؤ کی قدر 1 ایمپیئر ہوگی۔ 

فرض کریں کہ ایک الیکٹریکل سرکٹ میں 4 ایمپیئر کا فیوز ہے، اگر اس میں 4 ایمپیئر کا کرنٹ بہتا ہے تو یہ صحیح کام کرے گا، تاکہ اس میں کرنٹ کا دماغ 4 ایمپیئر سے بڑھ کر 20 ایمپیئر ہوجائے، تو فیوز استعمال کیا جائے 

 

ہمارے ٹی وی کی تصاویر 10mA کے برقی کرنٹ پر کام کرتی ہیں اور کی بورڈ اور ماؤس 50mA کے کرنٹ پر کام کرتے ہیں، جس کے لیے کرنٹ کی بہت کم قیمت کی ضرورت ہوتی ہے۔ 

 

جب کہ لیپ ٹاپ 3A استعمال کرتے ہیں اور ہمارے اسٹریم میں مائکروویو اوون 15A کی ندی پر کام کرتا ہے، جب کہ بادلوں کے ٹکرانے سے پیدا ہونے والی بجلی 10000A سے زیادہ ہے، اس لیے یہ بہت خطرناک ہے۔ 

1 ایمپیئر واٹ یا 1 ایمپیئر ٹو واٹ میں کتنے واٹ ہوتے ہیں۔ 

  1. 184 واٹ (AC) 
  1. یہ 230 واٹ (DC) میں ہے۔ 

آج کل امتحان میں ایمپیئر کی معیاری تعریف پوچھی جا رہی ہے جو کہ برقی مقناطیسی قوت کی بنیاد پر درج ذیل ہے۔ 

معیاری ایمپیئر کی تعریف : -  جب ویکیوم میں 1 میٹر کے فاصلے پر رکھے گئے دو کنڈکٹرز میں 1 ایمپیئر کا برقی کرنٹ لگایا جاتا ہے تو کنڈکٹرز 2 × 10 ∆ ⁷ N فی یونٹ لمبائی کی کشش یا پسپائی کی قوت پیدا کرتے ہیں۔"ہم اسے معیاری ایمپیئر کہتے ہیں ۔ 

 ایمپیئر کا فارمولا ایمپیئر = کولمب/کولمب۔دوسرا ہے 

                       1 A = 1C/1 سیکنڈ 

واٹ، وولٹ، ایچ پی، یونٹ کیا ہے؟ 

کرنٹ کو سمجھنے کے دوران ایمپیئر کے ساتھ ساتھ واٹ، وولٹ، یونٹ کو بھی سمجھنا ضروری ہے۔ 

وولٹ کیا ہے؟ 

وولٹ:-  وولٹیج کی اکائی وولٹ ہے۔ 

ایک وولٹ کی تعریف: - جب  1 اے کا کرنٹ 1 اوہم کی مزاحمت سے بہتا ہے تو دو پوائنٹس کے درمیان پیدا ہونے والے وولٹیج کا دماغ 1 وولٹ ہوگا۔چونکہ ہم جانتے ہیں کہ V = IR ( V = موجودہ × مزاحمت) 

1 واٹ کیا ہے؟ 

برقی طاقت کا یونٹ واٹ ہے۔ 

پاور = وولٹ × ایمپیئر 

1 واٹ = 1 جول فی سیکنڈ 

1 HP = 746 واٹ 

1 میٹرک HP = 735.5 واٹس 

1 یونٹ = 1 کلو واٹ گھنٹہ 

1 ایمپس سے واٹ 

amps to kw اور kw to amps کے دماغ کو تلاش کرنے کے لیے فارمولہ دیا گیا ہے تاکہ ہم ایمپیئر، kw، وولٹیج، سنگل فیز میں مزاحمت اور تھری فیز پاور سپلائی کی قدر معلوم کر سکیں۔ 

1 کلو واٹ میں کتنا ایمپیئر ہے؟ 

سنگل فیز سپلائی کے لیے 1 ایمپیئر تا واٹ فارمولا 

1 KW سے Amp :- 

تھری فیز موٹر میں 1 KW = 1.5 HP = 2.2 AMP ہے۔ 

تین فیز سپلائی کے لیے ایمپیئر فارمولا 

DC کے لیے 1 ایمپیئر سے واٹ 

واٹ = ایمپس ایکس وولٹ ڈی سی سپلائی ایمپیئر اور وولٹ کو ضرب دینے پر واٹ کی قدر حاصل کی جاتی ہے۔ 

جب یہاں وولٹیج کی قدر بڑھتی ہے تو ایمپیئر کی قدر کم ہو جاتی ہے اور جب وولٹیج کی قدر کم ہوتی ہے تو ایمپیئر کی قدر بڑھ جاتی ہے۔ 

فرض کریں ایمپیئر کی ویلیو 4A ہے اور وولٹ کی ویلیو 5V ہے تو واٹ کا دماغ 20W ہوگا۔ 

متبادل کرنٹ کے لیے 1 ایمپیئر تا واٹ 

واحد مرحلے کے لیے - 

واٹ = ایمپس ایکس وولٹ ایکس پی ایف 

جہاں پی ایف کو پاور فیکٹر کہا جاتا ہے۔ 

ایمپیئر، وولٹ اور واٹ میں کیا فرق ہے؟ 

آئیے 1 ایمپیئر سے واٹ کے ساتھ ایمپیئر، وولٹ اور واٹ کے درمیان فرق کو سمجھتے ہیں۔ 

ایمپیئر: - یہ برقی رو کی پیمائش کرنے کی اکائی ہے، اسے A سے ظاہر کیا جاتا ہے، اس کی قیمت ایک سیکنڈ میں بہنے والے چارجز کے برابر ہے۔ 

وولٹ دو پوائنٹس کے درمیان فرق کو ماپنے کی اکائی ہے، جسے V کے ذریعے دکھایا گیا ہے، جسے وولٹا نے دریافت کیا ہے۔ یہ ایک قسم کا دباؤ ہے جو الیکٹران کو دھکیلتا ہے۔ 

واٹ:- یہ طاقت کا SI یونٹ ہے۔اس کی قدر توانائی میں تبدیلی کی شرح کے برابر ہے۔
