كود اللون الذهبي RGB

رمز اللون الذهبي RGB = # FFD700 = 255 * 65536 + 215 * 256 + 0 = (255 ، 215 ، 0)

الأحمر = 255 ، والأخضر = 215 ، والأزرق = 0
